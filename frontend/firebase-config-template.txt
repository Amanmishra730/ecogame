# Firebase Configuration Template
# Copy this content to a new file called .env.local in the frontend directory

VITE_FIREBASE_API_KEY=your-actual-api-key-here
VITE_FIREBASE_AUTH_DOMAIN=your-project-id.firebaseapp.com
VITE_FIREBASE_PROJECT_ID=your-project-id
VITE_FIREBASE_STORAGE_BUCKET=your-project-id.appspot.com
VITE_FIREBASE_MESSAGING_SENDER_ID=your-sender-id
VITE_FIREBASE_APP_ID=your-app-id

# Steps to configure:
# 1. Go to https://console.firebase.google.com/
# 2. Create a new project or select existing one
# 3. Go to Project Settings > General > Your apps
# 4. Add a web app and copy the config values
# 5. Create a file called .env.local in the frontend directory
# 6. Copy the content above and replace with your actual values
# 7. Enable Authentication > Sign-in method > Email/Password
# 8. Restart your development server: npm run dev
